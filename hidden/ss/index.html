服务器快速部署

 1> zsh 安装
 2
 3apt install zsh -y;
 4# zsh-syntax-highlighting
 5sh -c "$(curl -fsSL https://raw.github.com/ohmyzsh/ohmyzsh/master/tools/install.sh)";
 6git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting;
 7# zsh-autosuggestions
 8git clone git://github.com/zsh-users/zsh-autosuggestions $ZSH_CUSTOM/plugins/zsh-autosuggestions;
 9
10plugins=(git zsh-syntax-highlighting zsh-autosuggestions) # 主题 ys
11ZSH_THEME="ys"
12
13chsh -s /bin/zsh
